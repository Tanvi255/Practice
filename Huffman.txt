import java.util.*;

class Node implements Comparable<Node>{
	int freq;
	String symbol;
	Node left,right;
	String huff;

	Node(int freq,String symbol,Node left,Node right){
		this.freq = freq;
		this.symbol = symbol;
		this.left = left;
		this.right = right;
		this.huff = "";
	}

	public int compareTo(Node n){
		return this.freq-n.freq;
	}
}

class Demo{
	static void printCode(Node node , String val){
		String newVal = val + node.huff;

		if(node.left != null)
			printCode(node.left,newVal);
		if(node.right != null)
			printCode(node.right,newVal);

		if(node.left == null && node.right == null)
			System.out.println(node.symbol + "->" + newVal);
	}
	public static void main(String [] args){
		Scanner sc = new Scanner(System.in);

		System.out.println("Enter no. of char ");
		int n = sc.nextInt();

		String charA[] = new String[n];
		int freq[] = new int[n];

		for(int i = 0 ;i <n ; i++){
			System.out.println("Enter char");
			charA[i] = sc.next();
			System.out.println("Enter freq");
			freq[i] = sc.nextInt();
		}

		PriorityQueue<Node> pq =new PriorityQueue<>();

		for(int i =0 ; i<n ;i++){
			pq.add(new Node(freq[i],charA[i],null,null));
		}

		while(pq.size()>1){
			Node left = pq.poll();
			Node right = pq.poll();

			left.huff = "0";
			right.huff = "1";

			Node newNd = new Node(left.freq+right.freq , left.symbol+right.symbol , left , right);

			pq.add(newNd);
		}

		System.out.println("Code for char are : ");
		printCode(pq.peek(),"");
	}
}
